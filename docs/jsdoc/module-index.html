(inner) requireAuth(allowedRole) → {function}

Description:

  • Checks if user is authenticated and has required role. Managers bypass role checks.

Source:
  • index.js, line 239
Examples
// Require cashier role
app.get('/cashier', requireAuth('cashier'), (req, res) => {...});
// Require authentication only (any role)
app.get('/profile', requireAuth(), (req, res) => {...});
// Allow multiple roles
app.get('/orders', requireAuth(['cashier', 'cook']), (req, res) => {...});
